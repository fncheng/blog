## 生成表格模拟数据

```ts
const generateData = (count: number) => {
    const data = []
    for (let i = 0; i < count; i++) {
        data.push({
            key: i.toString(),
            name: `姓名${i + 1}`,
            age: 20 + (i % 50),
            address: `西湖区湖底公园${i + 1}号`
        })
    }
    return data
}

const dataSource = generateData(100)

const columns = [
    {
        title: '姓名',
        dataIndex: 'name',
        key: 'name'
    },
    {
        title: '年龄',
        dataIndex: 'age',
        key: 'age'
    },
    {
        title: '住址',
        dataIndex: 'address',
        key: 'address'
    }
]
```
