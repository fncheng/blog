深度思考可折叠组件

ThinkCollapsible.vue

```vue
<template>
  <div v-if="thinkContent" class="think-collapsible">
    <div 
      class="think-header" 
      @click="toggleThinkCollapse"
      :class="{ 'collapsed': isCollapsed }"
    >
      <div class="think-header-left">
        <div class="think-icon" :class="{ 'thinking': isThinking }">
          <el-icon v-if="isThinking"><Loading /></el-icon>
          <el-icon v-else><Check /></el-icon>
        </div>
        <span class="think-text">{{ isThinking ? '思考中' : `已深度思考 (用时${duration}秒)` }}</span>
      </div>
      <div class="think-arrow">
        <el-icon><ArrowUp /></el-icon>
      </div>
    </div>
    <div 
      v-show="!isCollapsed" 
      class="think-content"
    >
      <p class="think-content-text">{{ thinkContent }}</p>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, watch } from 'vue'
import { Check, ArrowUp, Loading } from '@element-plus/icons-vue'

interface Props {
  thinkContent: string
  isThinking?: boolean
  duration?: number
}

const props = withDefaults(defineProps<Props>(), {
  isThinking: false,
  duration: 0
})

const isCollapsed = ref(false)

const toggleThinkCollapse = () => {
  isCollapsed.value = !isCollapsed.value
}

// 当思考内容变化时，自动展开
watch(() => props.thinkContent, (newContent) => {
  if (newContent && !isCollapsed.value) {
    isCollapsed.value = false
  }
})
</script>
```
